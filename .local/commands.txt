# Condiciones para que una repo pueda ser añadida como submodule:
    - Estar en un repositorio Git (por ejemplo, GitHub)
    - Ser accesible (tener permisos de lectura)
    - Tener al menos un commit

# Pasos para añadir como submodule:
    # Descargar el submodulo en la raiz
      git submodule add https://github.com/ErickaRamosBjumper/api-contract.git
    
    # Confirmar con un commit:
      git add .
      git commit -m "added submódulo api-contract"
      git push
    



# Descargar el submodule en una carpeta
git submodule add <URL-del-repositorio> <directorio-donde-colocarlo>
git submodule add  https://github.com/ErickaRamosBjumper/api-contract.git git_modules

git submodule update --init

# Actualizar submodulos
git submodule update

# Clonar un repositorio con submódulos
git clone --recurse-submodules <URL-del-repositorio-principal>

# Inicializar submódulos en un repositorio ya clonado
git submodule init
git submodule update

# Revisa si existen submodulos
git submodule status | cat
